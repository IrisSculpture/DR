country_event = {
	id = ger_con.1
	title = ger_con.1.t
	desc = ger_con.1.d
	picture = GFX_ger_con_uni_pict
	is_triggered_only = yes
	option = {
		name = ger_con.1.a
		every_state = {
			limit = {
				is_core_of = ROOT
			}
			add_core_of = WWS
		}
		WWS = { annex_country = { target = ROOT transfer_troops = yes } }
		ROOT = {
			every_unit_leader = {
				set_nationality = WWS
			}
		}
	}
	option = {
		name = ger_con.1.b
		trigger = {
			check_variable = { ger_support < 40 }
			OR = {
				check_variable = { pre_support < 90 }
				check_variable = { aus_support > 30 }
			}
			NOT = {
				OR = {
					tag = AGC
					tag = WWS
					tag = AOX
				}
			}
		}
		every_state = {
			limit = {
				is_core_of = ROOT
			}
			add_core_of = WWS
		}
		WWS = {  
			create_wargoal = {
				type = annex_everything
				target = ROOT
			}
		}
	}
}
