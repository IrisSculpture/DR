scripted_gui = {

	DR_money_system = {
		window_name = "money_system_ui_window"
		context_type =player_context
		parent_window_token = politics_tab
		
		
		visible = {
			always = yes
		}
		effects = {
			money_system_button_click={ 
				set_custom_slider = yes
				if={
					limit={
						NOT = {
							has_country_flag = entry_money_system_ui_window
						}
					}
					set_country_flag=entry_money_system_ui_window
				}
				else={
					clr_country_flag=entry_money_system_ui_window
				}
			}
			
		} 
	}
	DR_money_system_main = {
		window_name = "money_system_main_window"
		context_type =player_context
		# parent_window_token = politics_tab
		properties={
			
	    }
		visible = {
			has_country_flag = entry_money_system_ui_window
		} 
		effects = {
			
			
		}
		dynamic_lists = {
			custom_slider_grid = {
				array = slider_buttons
				change_scope = no
				entry_container = custom_slider_container
				index = i
				value = v
			}

			parliament_grid = {
				array = parliament_seats
				change_scope = no
				entry_container = parliament_seat_container
				index = idx
			}
		}

		properties = {
			slider_dot = {
				x = slider_dot_x
			}
			custom_slider_icon = {
				frame = button_frame
			}
		}
		effects = {
			gdp_icon_click = {
				 if = {
					 limit = { NOT = { has_variable = gdp_table_open } }
					 open_table = yes
				 }
			}

			custom_slider_icon_click = {
				set_variable = { current_int = v }
				change_parliament_seats = yes
				set_variable = { people_tax_qaq = current_int }
				multiply_variable = { people_tax_qaq = 2 }
				set_variable = { people_tax = people_tax_qaq }
				multiply_variable = { people_tax = 0.01 }
			}

			slider_limit_left_click = {
				add_to_variable = { current_int = -2 }
				clamp_variable = {
					var = current_int
					min = 0
					max = 50
				}
				change_parliament_seats = yes
				set_variable = { people_tax_qaq = current_int }
				multiply_variable = { people_tax_qaq = 2 }
				set_variable = { people_tax = people_tax_qaq }
				multiply_variable = { people_tax = 0.01 }
			}
			slider_limit_right_click = {
				add_to_variable = { current_int = 2 }
				clamp_variable = {
					var = current_int
					min = 0
					max = 50
				}
				change_parliament_seats = yes
				set_variable = { people_tax_qaq = current_int }
				multiply_variable = { people_tax_qaq = 2 }
				set_variable = { people_tax = people_tax_qaq }
				multiply_variable = { people_tax = 0.01 }
			}

		}
		triggers = {
			custom_slider_icon_visible = {
				if = {
					limit = {
						check_variable = {
							current_int = v
						}
					}
					set_temp_variable = { button_frame =  2 }
				}
				else = {
					set_temp_variable = { button_frame =  1 }
				}
			}
		}
	}
}
